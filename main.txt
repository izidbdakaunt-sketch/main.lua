-- === КОРРЕКТНЫЙ 2D ESP С ОЧИСТКОЙ ===
local ESP_Objects = {}

local function RemoveESP(player)
    if ESP_Objects[player] then
        for _, obj in pairs(ESP_Objects[player]) do
            if obj.Remove then obj:Remove() end
        end
        ESP_Objects[player] = nil
    end
end

local function CreateDrawingESP(player)
    if player == LocalPlayer then return end
    
    local Box = Drawing.new("Square"); Box.Thickness = 1; Box.Filled = false; Box.Color = Color3.new(1, 1, 1); Box.Visible = false
    local HealthOutline = Drawing.new("Square"); HealthOutline.Thickness = 1; HealthOutline.Filled = true; HealthOutline.Color = Color3.new(0, 0, 0); HealthOutline.Visible = false
    local HealthBar = Drawing.new("Square"); HealthBar.Thickness = 1; HealthBar.Filled = true; HealthBar.Visible = false
    local Name = Drawing.new("Text"); Name.Size = 13; Name.Center = true; Name.Outline = true; Name.Color = Color3.new(1,1,1); Name.Visible = false

    ESP_Objects[player] = {Box = Box, HealthOutline = HealthOutline, HealthBar = HealthBar, Name = Name}

    local conn
    conn = RunService.RenderStepped:Connect(function()
        local char = player.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        
        if Config.ESP and char and hum and hrp and hum.Health > 0 and char.Parent then
            local pos, onS = Camera:WorldToViewportPoint(hrp.Position)
            
            if onS then
                local sizeY = (Camera:WorldToViewportPoint(hrp.Position + Vector3.new(0, 3, 0)).Y - Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0)).Y)
                local sizeX = sizeY / 2
                local boxX, boxY = pos.X - (sizeX / 2), pos.Y - (sizeY / 2)
                
                Box.Position = Vector2.new(boxX, boxY); Box.Size = Vector2.new(sizeX, sizeY); Box.Visible = true
                
                local healthPercent = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                local barX, barY, barW, barH = boxX - 6, boxY, 3, sizeY
                
                HealthOutline.Position = Vector2.new(barX, barY); HealthOutline.Size = Vector2.new(barW, barH); HealthOutline.Visible = true
                
                local currentBarH = barH * healthPercent
                HealthBar.Position = Vector2.new(barX, barY + (barH - currentBarH))
                HealthBar.Size = Vector2.new(barW, currentBarH)
                HealthBar.Color = Color3.fromHSV(healthPercent * 0.35, 1, 1)
                HealthBar.Visible = true
                
                Name.Position = Vector2.new(pos.X, boxY - 15); Name.Text = player.Name; Name.Visible = true
                return
            end
        end
        Box.Visible = false; HealthOutline.Visible = false; HealthBar.Visible = false; Name.Visible = false
    end)
    
    -- Очистка при смерти или выходе
    player.CharacterRemoving:Connect(function() 
        Box.Visible = false; HealthOutline.Visible = false; HealthBar.Visible = false; Name.Visible = false 
    end)
end

Players.PlayerRemoving:Connect(RemoveESP)
for _, p in pairs(Players:GetPlayers()) do CreateDrawingESP(p) end
Players.PlayerAdded:Connect(CreateDrawingESP)

    btn.Text = text; btn.TextColor3 = Color3.fromRGB(255, 255, 255); btn.Font = Enum.Font.RobotoMono; btn.TextSize = 12; btn.Parent = parent; Instance.new("UICorner", btn)

end

local function createSlider(name, parent, order)
    local f = Instance.new("Frame", parent); f.Size = UDim2.new(1, 0, 0, 55); f.Position = UDim2.new(0, 0, 0, (order-1)*62); f.BackgroundColor3 = Color3.fromRGB(25, 25, 25); Instance.new("UICorner", f)
    local l = Instance.new("TextLabel", f); l.Size = UDim2.new(1, 0, 0, 25); l.BackgroundTransparency = 1; l.Text = name; l.TextColor3 = Color3.fromRGB(255, 255, 255); l.Font = Enum.Font.RobotoMono; l.TextSize = 11
    local b = Instance.new("Frame", f); b.Size = UDim2.new(0.85, 0, 0, 4); b.Position = UDim2.new(0.075, 0, 0.7, 0); b.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
    local d = Instance.new("Frame", b); d.Size = UDim2.new(0, 10, 0, 10); d.Position = UDim2.new(0.3, -5, 0.5, -5); d.BackgroundColor3 = Color3.fromRGB(255, 255, 255); Instance.new("UICorner", d).CornerRadius = UDim.new(1,0)
    local s = false
    f.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then s = true end end)
    UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then s = false end end)
    return {bar = b, dot = d, label = l, getSliding = function() return s end}
end

local aimFovS = createSlider("AIM FOV: 150", RightSide, 1)
local smoothS = createSlider("SMOOTHNESS: 0.15", RightSide, 2)
local camFovS = createSlider("CAMERA FOV: 70", RightSide, 3)

local FOVCircle = Drawing.new("Circle")
FOVCircle.Thickness = 1.5; FOVCircle.Radius = Config.AimFOV; FOVCircle.Visible = false; FOVCircle.Color = Color3.new(1,1,1)

RunService.RenderStepped:Connect(function()
    if aimFovS.getSliding() then
        local x = math.clamp((Mouse.X - aimFovS.bar.AbsolutePosition.X) / aimFovS.bar.AbsoluteSize.X, 0, 1)
        aimFovS.dot.Position = UDim2.new(x, -5, 0.5, -5); Config.AimFOV = math.floor(x * 500) + 50; aimFovS.label.Text = "AIM FOV: " .. Config.AimFOV; FOVCircle.Radius = Config.AimFOV
    elseif smoothS.getSliding() then
        local x = math.clamp((Mouse.X - smoothS.bar.AbsolutePosition.X) / smoothS.bar.AbsoluteSize.X, 0, 1)
        smoothS.dot.Position = UDim2.new(x, -5, 0.5, -5); Config.Smoothness = (x * 0.95) + 0.05; smoothS.label.Text = "SMOOTHNESS: " .. string.format("%.2f", Config.Smoothness)
    elseif camFovS.getSliding() then
        local x = math.clamp((Mouse.X - camFovS.bar.AbsolutePosition.X) / camFovS.bar.AbsoluteSize.X, 0, 1)
        camFovS.dot.Position = UDim2.new(x, -5, 0.5, -5); Config.CameraFOV = math.floor(x * 50) + 70; camFovS.label.Text = "CAMERA FOV: " .. Config.CameraFOV; Camera.FieldOfView = Config.CameraFOV
    end
    FOVCircle.Position = Vector2.new(Mouse.X, Mouse.Y + GuiService:GetGuiInset().Y); FOVCircle.Visible = Config.Aimlock
    
    if Config.Aimlock and UIS:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
        local target, dist = nil, Config.AimFOV
        for _, v in pairs(Players:GetPlayers()) do
            if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild(Config.AimPart) then
                local pos, onS = Camera:WorldToViewportPoint(v.Character[Config.AimPart].Position)
                if onS then local mD = (Vector2.new(pos.X, pos.Y) - Vector2.new(Mouse.X, Mouse.Y)).Magnitude
                if mD < dist then dist = mD; target = v end end
            end
        end
        if target then Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, target.Character[Config.AimPart].Position), Config.Smoothness) end
    end
end)
