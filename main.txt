local function Decode(data)
    local b = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
    data = string.gsub(data, '[^'..b..'=]', '')
    return (data:gsub('.', function(x)
        if (x == '=') then return '' end
        local r, f = '', (b:find(x) - 1)
        for i = 6, 1, -1 do r = r .. (f % 2^i - f % 2^(i - 1) > 0 and '1' or '0') end
        return r;
    end):gsub('%d%d%d%d%d%d%d%d', function(x)
        local c = 0
        for i = 1, 8 do c = c + (x:sub(i, i) == '1' and 2^(8 - i) or 0) end
        return string.char(c)
    end))
end

    local function IsVisible(part)
        if not Config.WallCheck then return true end
        local p = RaycastParams.new(); p.FilterType = Enum.RaycastFilterType.Exclude; p.FilterDescendantsInstances = {LocalPlayer.Character, part.Parent}
        local r = workspace:Raycast(Camera.CFrame.Position, part.Position - Camera.CFrame.Position, p)
        return r == nil
    end

        if Config.FullBright then Lighting.Brightness = 2; Lighting.ClockTime = 14; Lighting.Ambient = Color3.new(1,1,1); Lighting.GlobalShadows = false end
        Camera.FieldOfView = Config.CameraFOV
        
        local sliders = {aimFovS, smoothS, camFovS, predS}
        for _, sl in pairs(sliders) do
            if sl.getS() then
                local x = math.clamp((Mouse.X - sl.bar.AbsolutePosition.X) / sl.bar.AbsoluteSize.X, 0, 1)
                sl.dot.Position = UDim2.new(x, -8, 0.5, -8)
                local val = sl.min + (x * (sl.max - sl.min))
                if sl == aimFovS then Config.AimFOV = val elseif sl == smoothS then Config.Smoothness = val/100 elseif sl == camFovS then Config.CameraFOV = val elseif sl == predS then Config.Prediction = val/1000 end
                sl.label.Text = sl.name .. ": " .. math.floor(val)
            end
        end
        
        FOVCircle.Radius = Config.AimFOV; FOVCircle.Position = Vector2.new(Mouse.X, Mouse.Y + GuiService:GetGuiInset().Y); FOVCircle.Visible = Config.Aimlock

                for _, v in pairs(Players:GetPlayers()) do
                    if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild(Config.AimPart) then

                            if onS and IsVisible(part) then
                                local mDist = (Vector2.new(pos.X, pos.Y) - Vector2.new(Mouse.X, Mouse.Y)).Magnitude
                                if mDist < dist then dist = mDist; LockedTarget = part end
                            end
                        end
                    end
                end
            end
            if LockedTarget and LockedTarget.Parent then
                local predPos = LockedTarget.Position + (LockedTarget.Velocity * Config.Prediction)
                Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, predPos), Config.Smoothness)

    end)
